### `src/theme/useTheme.ts`

```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

type Theme = 'dark' | 'light' | 'system';

interface ThemeStore {
  theme: Theme;
  setTheme: (t: Theme) => void;
}

export const useThemeStore = create<ThemeStore>()(
  persist(
    (set) => ({
      theme: 'system',
      setTheme: (theme) => set({ theme }),
    }),
    { name: 'mynah-theme' },
  ),
);

export function resolveTheme(t: Theme): 'dark' | 'light' {
  if (t !== 'system') return t;
  return window.matchMedia('(prefers-color-scheme: light)').matches ? 'light' : 'dark';
}
```

### `src/theme/ThemeProvider.tsx`

```tsx
import type { JSX, ReactNode } from 'react';
import { useEffect } from 'react';
import { resolveTheme, useThemeStore } from './useTheme';

export function ThemeProvider({ children }: { children: ReactNode }): JSX.Element {
  const theme = useThemeStore((s) => s.theme);

  useEffect(() => {
    const apply = (): void => {
      const resolved = resolveTheme(theme);
      document.documentElement.setAttribute('data-theme', resolved);
    };
    apply();
    if (theme === 'system') {
      const mql = window.matchMedia('(prefers-color-scheme: light)');
      mql.addEventListener('change', apply);
      return () => mql.removeEventListener('change', apply);
    }
    return undefined;
  }, [theme]);

  return <>{children}</>;
}
```

### `src/components/shell/ProductSwitcher.tsx`

Accessible glass-styled select, native `<select>` under the hood — no custom popover yet. Future can upgrade to Radix `<Select>` once we need search.

```tsx
import type { JSX } from 'react';
import { useEffect } from 'react';
import { ChevronsUpDown } from 'lucide-react';
import { useProducts } from '../../hooks/useProducts';
import { useProductStore } from '../../stores/product.store';
import { cn } from '../../lib/cn';

export function ProductSwitcher(): JSX.Element {
  const products = useProducts();
  const selected = useProductStore((s) => s.selectedProductId);
  const setSelected = useProductStore((s) => s.setSelectedProductId);

  // Auto-pick first product when nothing selected
  useEffect(() => {
    if (!selected && products.data && products.data.length > 0) {
      setSelected(products.data[0]!.id);
    }
  }, [selected, products.data, setSelected]);

  if (products.isLoading) {
    return (
      <div className="glass flex h-9 items-center gap-2 px-3 text-sm text-[var(--fg-muted)]">
        Loading products…
      </div>
    );
  }
  const items = products.data ?? [];
  if (items.length === 0) {
    return (
      <div className="glass flex h-9 items-center gap-2 px-3 text-sm text-[var(--fg-muted)]">
        No products yet
      </div>
    );
  }

  return (
    <label
      className={cn(
        'glass glass-hover flex h-9 items-center gap-2 px-3 text-sm cursor-pointer',
        'focus-within:ring-2 focus-within:ring-[var(--accent-ring)]',
      )}
    >
      <select
        value={selected ?? ''}
        onChange={(e) => setSelected(e.target.value)}
        className="bg-transparent outline-none appearance-none pr-6 text-[var(--fg)]"
      >
        {items.map((p) => (
          <option key={p.id} value={p.id} className="bg-[var(--bg-elevated)] text-[var(--fg)]">
            {p.name}
          </option>
        ))}
      </select>
      <ChevronsUpDown size={16} className="text-[var(--fg-muted)]" />
    </label>
  );
}
```

Data sources:

- `costMillicentsToday` = sum of the same
- `status` = `'ready'` by default; `'running'` if any row in last 60s; `'error'` if the last row's `quality_score` indicates failure (we don't track per-call success today; default `ready`).

```tsx
function StatusDot({ status }: { status: 'ready' | 'running' | 'idle' | 'error' }): JSX.Element {
  const color =
    status === 'ready'    ? 'bg-emerald-400'
    : status === 'running' ? 'bg-[var(--color-accent)]'
    : status === 'error'   ? 'bg-red-400'
    :                        'bg-[var(--fg-subtle)]';
  return <span className={`inline-block h-2 w-2 rounded-full ${color}`} />;
}
```
